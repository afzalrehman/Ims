    function toggleFullScreen() {
      const elem = document.documentElement;

      if (!document.fullscreenElement && !document.webkitFullscreenElement && !document.msFullscreenElement) {
        // Element is not in full screen mode
        if (elem.requestFullscreen) {
          elem.requestFullscreen();
        } else if (elem.webkitRequestFullscreen) { /* Safari */
          elem.webkitRequestFullscreen();
        } else if (elem.msRequestFullscreen) { /* IE11 */
          elem.msRequestFullscreen();
        }
      } else {
        // Element is in full screen mode, so exit
        if (document.exitFullscreen) {
          document.exitFullscreen();
        } else if (document.webkitExitFullscreen) { /* Safari */
          document.webkitExitFullscreen();
        } else if (document.msExitFullscreen) { /* IE11 */
          document.msExitFullscreen();
        }
      }

                Fath_Name: Fath_Name.value,
                Gua_Phone_Number: Gua_Phone_Number.value,
                Shooba: Shooba.value
            },
            dataType: 'json',
            success: function(data) {
                let count = 1;
                $.each(data, function(index, row) {
                  table_body.innerHTML += `
                    <tr>
                        <td class="fw-semibold">`+count+`</td> <!-- Count -->
                        <td style="display:none;">`+row.id+`</td> <!-- ID -->
                        <td class="inter fw-semibold" >`+row.Registrasion_Number+`</td> <!-- Registration Number -->
                        <td class="urdu-menu">`+row.Student_Name+`</td> <!-- Student Name -->
                        <td class="urdu-menu">`+row.Father_Name+`</td> <!-- Student Father Name -->
                        <td class="fw-semibold">`+row.Guadian_Phone_Number+`</td> <!-- Phone Number -->
                        <td class="urdu-menu">`+row.Shooba+`</td> <!-- Shooba -->
                        <td>
                            <i class="fa-solid fa-eye text-primary mx-2"></i> <!-- View Data -->
                            <i class="fa-solid fa-edit text-success mx-2" data-bs-toggle="modal"
                                data-bs-target="#largeModal" onclick="Edit_User(`+row.id+`)"></i> <!-- Edit Data -->
                            <i class="fa-solid fa-trash-can text-danger mx-2" onclick="Delete_User(`+row.id+`)"></i> <!-- Delete Data -->
                        </td>
                    </tr>
                  `;
                  count++;
                });
            },
            error: function(xhr, status, error) {
                console.error(xhr.responseText);
            }
        });
    }

    //------------------------------- Delete Data -------------------------------//
    function Delete_User(id){
        if(confirm("Are you sure you want to delete this row?")){
            $.ajax({
                url: 'student-function.php',
                type: 'POST',
                data: {action: 'delete', id: id},
                success: function(response) {
                    console.log(response);
                    location.reload(); // Refresh the page to reflect changes
                },
                error: function(xhr, status, error) {
                    console.error(xhr.responseText);
                }
            });
        }
    }
